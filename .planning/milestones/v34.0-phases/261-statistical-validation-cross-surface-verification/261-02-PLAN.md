```javascript
      let priorityTotal = 0;
      let baselineTotal = 0;
      let conditionedDraws = 0;

        const entropy = rng();

        // Sample 4 trait colors under the heavy-tail distribution.
        const colors = [];

          const rnd32 = (rng() & 0xFFFFFFFFn);
          colors.push(jsWeightedColorBucket(rnd32));

        // baseline: rotation only — soloIdx_baseline = uint8((3 - (entropy & 3)) & 3)
        // priority: gold-priority — soloIdx_priority = goldQuadrant

        // shareBps at this offset (rotated): shares[i] = shareBps[(i + rotation + 1) & 3]
        //   (matches JackpotBucketLib.rotatedShareBps formula at L248-250).
        // payoutPerTicket[quadrantIdx] = shares[quadrantIdx] / counts[quadrantIdx] (in BPS, then / 10000 for fraction).
        function payoutPerTicket(quadrantSoloIdx) {
          // Helper: under the assumption that quadrantSoloIdx receives the solo
          // bucket assignment, what's the EV for the holder of the gold-color
          // winning trait at quadrant `goldQuads[0]` (we conditioned on ≥1 gold —
          // the holder owns *the* gold-color winning quadrant; per D-07 we use the
          // first gold quadrant in the array as the holder's quadrant — the
          // priority model puts probability 1/goldCount on each gold quadrant
          // becoming solo, so we compute EV by averaging over goldQuads).
          // Here we evaluate "what does the holder receive if quadrant Q is the
          // solo bucket?" — see the calling code below.
          return 0; // placeholder — actual logic inlined below

        // Under the priority model, the gold-color winning quadrant becomes solo
        // with probability 1/goldCount (each of the goldQuads is equally likely
        // by uniform tie-break). The holder of any one gold winning trait is at
        // some specific quadrant g ∈ goldQuads. Their EV averaged over the
        // tie-break randomness = 1/goldCount * (solo-share / 1) + (goldCount-1)/goldCount * E[non-solo-share-at-g].
        // E[non-solo-share-at-g] averages share/count over the 3 non-solo bucket
        // assignments g could receive.
        //
        // Under the baseline model, soloIdx is rotation; the holder receives:
        //   if g == rotation: solo-share / 1
        //   else: shares[g_rotated] / counts[g_rotated]  (g sees 1 of the 3 non-solo assignments deterministically based on rotation)
        // Average over the rotation distribution to get baseline EV.
        //
        // Both EVs are CONDITIONED on the holder being at quadrant g (one of the
        // gold quadrants) — that's the "owns-the-gold-quadrant-ticket" model.

        const goldCount = goldQuads.length;
        const g = goldQuads[0]; // any gold quadrant; D-07 model is symmetric — pick first

        // Under PRIORITY (gold-priority):
        // P(g is solo) = 1/goldCount; P(g is non-solo at offset r) summed over the 3 non-solo offsets = (goldCount-1)/goldCount averaged uniformly over those.
        // Solo payout-per-ticket on this surface = surf.shareBps[soloBucketIdx] / counts[soloBucketIdx].
        // Solo bucket has count = 1; share index for solo bucket under offset is rotatedShareBps with traitIdx == soloBucketIdx... but at base counts the solo BUCKET is the one with count=1, and its shareBps is whichever element of surf.shareBps lands at that bucket under rotation.         //   counts[i]  = BASE_COUNTS[(i + rotation) & 3]
        //   shares[i]  = surf.shareBps[(i + rotation + 1) & 3]    // per rotatedShareBps L248-250
        //   payoutPerTicketAtQuadrantI(rotation) = shares[i] / counts[i]
        //
        // Under PRIORITY: rotation low-bits are still entropy & 3 (not modified by helper);
        // the SOLO ASSIGNMENT (which quadrant gets the solo-share) is goldQuadrant
        // (returned by tester). But the BUCKET COUNT and SHARE BPS arrays are computed
        // off effectiveEntropy = (entropy & ~3) | ((3 - goldQuadrant) & 3). So:
        //
        //   priorityRotation = (3 - goldQuadrant) & 3
        //   counts_priority[i]  = BASE_COUNTS[(i + priorityRotation) & 3]
        //   shares_priority[i]  = surf.shareBps[(i + priorityRotation + 1) & 3]
        //   priorityPayoutAtG = shares_priority[g] / counts_priority[g]

        // Under BASELINE (no priority — rotation = entropy & 3):
        //   baselineRotation = Number(entropy & 3n)
        //   counts_baseline[i]  = BASE_COUNTS[(i + baselineRotation) & 3]
        //   shares_baseline[i]  = surf.shareBps[(i + baselineRotation + 1) & 3]
        //   baselinePayoutAtG = shares_baseline[g] / counts_baseline[g]

        const priorityRotation = (3 - goldQuadrant) & 3;
        const baselineRotation = Number(entropy & 3n);

        function payoutAtQuadrant(rotation, quadrant) {
          const count = BASE_COUNTS[(quadrant + rotation) & 3];
          const share = surf.shareBps[(quadrant + rotation + 1) & 3];
          return share / count; // in BPS units; ratios cancel out the /10000 factor
        }

        priorityTotal += payoutAtQuadrant(priorityRotation, g);
        baselineTotal += payoutAtQuadrant(baselineRotation, g);
      }

      const priorityEV = priorityTotal / conditionedDraws;
      const baselineEV = baselineTotal / conditionedDraws;
      const measuredUplift = priorityEV / baselineEV;
      const lowerBound = surf.analyticalUplift * (1 - TOLERANCE_REL);
      const upperBound = surf.analyticalUplift * (1 + TOLERANCE_REL);

      console.log(`  ${surf.name}: measured uplift ${measuredUplift.toFixed(3)}× over ${conditionedDraws} conditioned draws (analytical ${surf.analyticalUplift}× ±5%)`);
      expect(measuredUplift, `measured ${measuredUplift.toFixed(3)} not in [${lowerBound.toFixed(3)}, ${upperBound.toFixed(3)}]`).to.be.gte(lowerBound);
      expect(measuredUplift).to.be.lte(upperBound);
    });
  }
});
```

**Wilson 99% CI helper (closed-form):**
```javascript
// Wilson score interval for binomial proportion, two-sided 99% (z = 2.576).
function wilson99(successes, total) {
  const z = 2.576;
  const p = successes / total;
  const denom = 1 + (z * z) / total;
  const center = (p + (z * z) / (2 * total)) / denom;
  const half = (z * Math.sqrt((p * (1 - p)) / total + (z * z) / (4 * total * total))) / denom;
  return { lo: center - half, hi: center + half };
}
```
