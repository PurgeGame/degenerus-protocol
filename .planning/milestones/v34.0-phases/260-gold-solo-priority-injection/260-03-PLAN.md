async function computeHash2(randWord, lvl) {
  // EntropyLib.hash2 implementation: keccak256(abi.encode(randWord, lvl))
  // Confirm the exact pattern by reading contracts/libraries/EntropyLib.sol.
  const coder = new hre.ethers.AbiCoder();
  const encoded = coder.encode(["uint256", "uint24"], [randWord, lvl]);
  const hash = hre.ethers.keccak256(encoded);
  return BigInt(hash);
}

async function rollAndUnpackTraits(randWord) {
  // Replicates _rollWinningTraits(randWord, false) and unpackWinningTraits off-chain.
  // _rollWinningTraits with isBonus=false uses DegenerusTraitUtils.packedTraitsFromSeed(randWord)
  // which composes traitFromWord across 4 64-bit lanes plus quadrant flags.
  // For GOLD_RANDWORD all four lanes have low 32 = 0xFFFFFFFF → color 7, symbol = (high 32) & 7.
  // Deploy a TraitUtilsTester instance and call packedTraitsFromSeed directly.
  const TraitTester = await hre.ethers.getContractFactory("TraitUtilsTester");
  const traitTester = await TraitTester.deploy();
  await traitTester.waitForDeployment();
  const packed = await traitTester.packedTraitsFromSeed(randWord);
  // unpack: 4 trait IDs, low byte first.
  const t0 = Number(packed) & 0xFF;
  const t1 = (Number(packed) >> 8) & 0xFF;
  const t2 = (Number(packed) >> 16) & 0xFF;
  const t3 = (Number(packed) >> 24) & 0xFF;
  return [t0, t1, t2, t3];
