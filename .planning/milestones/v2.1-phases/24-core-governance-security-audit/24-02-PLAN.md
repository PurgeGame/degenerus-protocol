function vote(uint256 proposalId, bool approve) external {
    // Stall re-check
    uint48 lastVrf = gameAdmin.lastVrfProcessed();
    if (block.timestamp - uint256(lastVrf) < ADMIN_STALL_THRESHOLD) revert NotStalled();

    Proposal storage p = proposals[proposalId];
    if (p.state != ProposalState.Active || p.createdAt == 0) revert ProposalNotActive();

    // Expiry check (168h)
    if (block.timestamp - uint256(p.createdAt) >= PROPOSAL_LIFETIME) {
        p.state = ProposalState.Expired;
        unchecked { activeProposalCount--; }
        revert ProposalExpired();
    }

    uint256 weight = sDGNRS.balanceOf(msg.sender);
    if (weight == 0) revert InsufficientStake();

    // Vote change: subtract old weight, add new
    Vote currentVote = votes[proposalId][msg.sender];
    if (currentVote != Vote.None) {
        uint256 oldWeight = voteWeight[proposalId][msg.sender];
        if (currentVote == Vote.Approve) { p.approveWeight -= oldWeight; }
        else { p.rejectWeight -= oldWeight; }

    voteWeight[proposalId][msg.sender] = weight;
    if (approve) { p.approveWeight += weight; }

   If voter has 100 sDGNRS, votes Approve (weight=100). Then someone transfers 50 sDGNRS to voter (now 150). Voter changes to Reject:
   - oldWeight = 100 (from previous vote), approveWeight -= 100 (correct subtraction)
   - weight = 150 (live balance), rejectWeight += 150
   - Net: approveWeight lost 100, rejectWeight gained 150. Delta = +50 net weight.
   - This is correct behavior IF sDGNRS is frozen during stall (VOTE-01). If sDGNRS is NOT frozen, this creates a weight amplification vector.

3. **Adversarial check -- double-counting via same-direction re-vote:**
   Voter votes Approve (100). Balance increases to 150. Voter votes Approve again:
   - currentVote=Approve, oldWeight=100, approveWeight -= 100

   - Net: approveWeight = original - 100 + 150 = original + 50
   - Correct -- re-voting in same direction updates to current balance, no double-count.

4. **Adversarial check -- underflow in subtraction:**
