    ```solidity
    /// @title RedemptionInvariants -- Proves gambling burn redemption system invariants
    /// @notice 7 invariants encoding Phase 44 corrected properties (INV-01 through INV-07)
    contract RedemptionInvariants is DeployProtocol {
        RedemptionHandler public handler;
        VRFHandler public vrfHandler;

        // Storage slot constants for internal sDGNRS state
        uint256 private constant SLOT_PENDING_BURNIE = 10;
        uint256 private constant SLOT_SUPPLY_SNAPSHOT = 13;
    ```

    **7 invariant functions:**

    **INV-01: invariant_ethSegregationSolvency**
    Segregated ETH never exceeds what the contract can cover. Use `assertGe` (not `assertEq`) because the contract may have more ETH than the segregated amount (from other deposits, game winnings).
    ```solidity
    function invariant_ethSegregationSolvency() public view {
        uint256 segregated = sdgnrs.pendingRedemptionEthValue();
        uint256 ethBal = address(sdgnrs).balance;
        // Conservative check: just ETH balance. stETH and claimable winnings
        // are additional backing, so ethBal alone may undercount.
        // But segregated should never exceed total accessible funds.
        // Use the broader check: ethBal + stETH balance
        uint256 stethBal = mockStETH.balanceOf(address(sdgnrs));
        assertGe(
            ethBal + stethBal,
            segregated,
            "INV-01: segregated ETH exceeds contract ETH+stETH balance"
        );
    }
    ```

    **INV-02: invariant_noDoubleClaim**
    Read the handler's ghost_doubleClaim counter. If it's > 0, a re-claim succeeded after a successful claim, meaning the claim was not deleted properly.
    ```solidity
    function invariant_noDoubleClaim() public view {
        assertEq(
            handler.ghost_doubleClaim(),
            0,
            "INV-02: double claim succeeded (claim not deleted before payout)"
        );
    }
    ```

    **INV-03: invariant_periodIndexMonotonic**
    Read the handler's ghost_periodIndexDecreased counter.
    ```solidity
    function invariant_periodIndexMonotonic() public view {
        assertEq(
            handler.ghost_periodIndexDecreased(),
            0,
            "INV-03: redemption period index decreased"
        );
    }
    ```

    **INV-04: invariant_supplyConsistency**
    totalSupply should equal initialSupply minus total burned. Note: this only tracks burns through the handler. Other paths (deterministic burns, pool burns) might also reduce supply. The handler tracks `ghost_totalBurned` for gambling burns only. The invariant is: `totalSupply <= initialSupply` and `totalSupply >= initialSupply - ghost_totalBurned` (other burn paths may reduce further) and `initialSupply - ghost_totalBurned >= totalSupply` (handler's burns definitely reduced supply).
    Actually simpler: `sdgnrs.totalSupply() == handler.ghost_initialSupply() - handler.ghost_totalBurned()` IF only the handler is burning. Since we only target the handler, this should hold exactly.
    ```solidity
    function invariant_supplyConsistency() public view {
        uint256 expected = handler.ghost_initialSupply() - handler.ghost_totalBurned();
        assertEq(
            sdgnrs.totalSupply(),
            expected,
            "INV-04: totalSupply != initialSupply - totalBurned"
        );
    }
    ```

    **INV-05: invariant_fiftyPercentCap**
    The 50% cap is enforced per period: `redemptionPeriodBurned <= redemptionPeriodSupplySnapshot / 2`. Read both via `vm.load`.
    ```solidity
    function invariant_fiftyPercentCap() public view {
        uint256 snapshot = uint256(vm.load(address(sdgnrs), bytes32(uint256(SLOT_SUPPLY_SNAPSHOT))));
        uint256 burned = uint256(vm.load(address(sdgnrs), bytes32(uint256(SLOT_PERIOD_BURNED))));
        if (snapshot > 0) {
            assertLe(
                burned,
                snapshot / 2,
                "INV-05: period burned exceeds 50% of supply snapshot"
            );
        }
    }
    ```

    **INV-06: invariant_rollBounds**

    ```solidity
    function invariant_rollBounds() public view {

            handler.ghost_rollOutOfBounds(),
            0,

    ```

    **INV-07: invariant_aggregateTracking**
    The `pendingRedemptionEthValue` should be >= 0 (it's uint256, always true) and bounded above by the ETH balance. More meaningfully: after claims, the segregated amount should decrease. Use a weaker form: `pendingRedemptionEthValue + ghost_totalEthClaimed >= some_lower_bound`. Actually, the simplest useful check: `ghost_totalEthClaimed <= ethBal_initial + deposits`, but we don't track deposits directly.

    Use this approach: verify the aggregate accounting identity. The handler only does gambling burns (not deterministic). After all burns and claims, the total ETH that was segregated should be reconcilable. Since we can't perfectly track the total-ever-segregated, use the dust bound approach from the research:
    - Read `pendingRedemptionEthValue` (remaining segregated)
    - The invariant: `pendingRedemptionEthValue` is consistent -- it should not be negative (underflow would revert) and should not exceed the contract balance. This is a subset of INV-01 but specifically focuses on the tracking variable.
    - Additionally check: `pendingRedemptionBurnie` (slot 10) should not exceed the BURNIE balance + claimable.

    ```solidity
    function invariant_aggregateTracking() public view {
        // ETH tracking: segregated <= balance + stETH (overlap with INV-01 but uses different ghost path)
        uint256 segregatedEth = sdgnrs.pendingRedemptionEthValue();
        uint256 ethBal = address(sdgnrs).balance;
        uint256 stethBal = mockStETH.balanceOf(address(sdgnrs));
        assertGe(
            ethBal + stethBal,
            segregatedEth,
            "INV-07: ETH aggregate tracking exceeds balance"
        );

        // BURNIE tracking: reserved <= available
        uint256 pendingBurnie = uint256(vm.load(address(sdgnrs), bytes32(uint256(SLOT_PENDING_BURNIE))));
        uint256 burnieBal = coin.balanceOf(address(sdgnrs));
        // Note: claimable coinflip BURNIE is also backing, but hard to read in view context.
        // Use the weaker check: pendingBurnie <= burnieBal + reasonable_upper_bound.
        // For now, just ensure pendingBurnie doesn't wildly exceed burnieBal.
        // The dust bound from Phase 44: O(N * 99) wei per period.
        // With 5 actors and ~256 runs, max dust ~ 5 * 99 * 256 = 126720 wei.
        // Use a generous 1 ether bound for safety (production dust is << 1 ether).
        if (pendingBurnie > 0) {
            assertGe(
                burnieBal + 1 ether,
                pendingBurnie,
                "INV-07: BURNIE aggregate tracking exceeds balance + tolerance"
            );
        }
    }
    ```
