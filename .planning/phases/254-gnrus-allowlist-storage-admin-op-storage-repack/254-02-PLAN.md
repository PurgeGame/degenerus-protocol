<!-- Canonical revert order per REQUIREMENTS.md ALW-03 + Success Criterion 3 (locked-slot guard before branch dispatch). -->

Order is load-bearing — Phase 256 tests assert each revert path; Phase 257 AUDIT-02-(h) verifies locked-slot lock-bypass is impossible by construction. Re-ordering would change observable revert behavior.

1. **Vault-owner gate** — `if (!vault.isVaultOwner(msg.sender)) revert Unauthorized();`
2. **InvalidSlot** — `if (slot >= MAX_ACTIVE_SLOTS) revert InvalidSlot();` (note: `slot >= 20` per ALW-03 verbatim — `MAX_ACTIVE_SLOTS == 20` so equivalent)
3. **RecipientIsContract** — `if (recipient != address(0) && recipient.code.length != 0) revert RecipientIsContract();`
4. **Locked-slot guard** — `if (slot < LOCKED_SLOTS && currentSlate[slot] != address(0)) revert SlotLocked();` — fires BEFORE branch dispatch so locked slots cannot be mutated via the pending queue either (Success Criterion 3 verbatim)
5. **Branch dispatch** — read `currentSlate[slot]` once into a local; branch on `address(0)` vs non-zero.
6. **Removal special case (instant-apply branch only)** — if `current == address(0)` AND `recipient == address(0)`: per ALW-03 the SlotAlreadyEmpty revert fires only when ALL THREE hold (current==0, recipient==0, no pending edit). If pending IS set, the call writes zero to pending — cancel-the-queued-add semantics (D-254-PENDING-01: "a new setCharity call on the same slot overwrites the pending value (no separate cancel API)").
7. **Cap check (both branches)** — compute the would-be future active bitmap from this write + already-pending edits, popcount, compare to `MAX_ACTIVE_SLOTS`. Revert `CapExceeded` if `> 20`.
8. **Apply branch** — instant-apply OR queue (mutually exclusive, determined by step 5).
9. **Emit event** — `CharityApplied` or `CharityQueued`.

**2. Implement `setCharity`.** Place under the new section banner. EXACT signature (per ALW-03 + D-254-CONTEXT-01 specifics; default to no return value per Claude's Discretion in CONTEXT.md):

```solidity
/// @notice Vault-owner-only single admin entry point covering charity slate add/replace/remove.
/// @dev Branches:
///      - Instant-apply (writes directly to current slate) when `currentSlate[slot] == address(0)`.
///      - Queue (writes to pending edit, applied at next pickCharity flush) when `currentSlate[slot] != address(0)`.
///      Pass `recipient == address(0)` to remove a slot. Locked slots (0/1/2) cannot be replaced or
///      removed once filled — first fill is an irrevocable commitment.
/// @param slot Slot index in the current slate (0..19)
/// @param recipient Charity recipient address (must be EOA when non-zero); pass address(0) to remove
function setCharity(uint8 slot, address recipient) external {
    // 1. Vault-owner gate
    if (!vault.isVaultOwner(msg.sender)) revert Unauthorized();

    // 2. InvalidSlot
    if (slot >= MAX_ACTIVE_SLOTS) revert InvalidSlot();

    // 3. RecipientIsContract (only when non-zero)
    if (recipient != address(0) && recipient.code.length != 0) revert RecipientIsContract();

    // 4. Locked-slot guard (BEFORE branch dispatch — slots 0/1/2 cannot be mutated through queue either)
    address current = currentSlate[slot];
    if (slot < LOCKED_SLOTS && current != address(0)) revert SlotLocked();

    uint32 slotMask = uint32(1) << slot;

    // 5. Branch dispatch
    if (current == address(0)) {
        // Instant-apply branch (or removal special case if recipient == 0)
        if (recipient == address(0)) {
            // Removal special case: empty current + zero recipient.
            // SlotAlreadyEmpty fires only when no pending edit exists for this slot.
            // If a pending edit IS set, this call cancels it: clear pendingEdit[slot] and the bitmap bit.
            if ((pendingEditSet & slotMask) == 0) revert SlotAlreadyEmpty();
            // Cancel the queued add: clear the pending entry and its bitmap bit.
            // A pending-add on an empty current slot is a no-op at flush, so removing it keeps pendingEditSet representing only state-mutating edits.
            pendingEdit[slot] = address(0);
            pendingEditSet &= ~slotMask;
            emit CharityQueued(slot, address(0));
            return;
        }
        // Cap check (instant-apply branch, recipient != 0): post-write currentActiveBitmap will have bit `slot` set.
        // Future bitmap = (currentActiveBitmap | slotMask) modified by pending edits as if flush ran.
        uint32 futureBitmap = _futureBitmapAfter(slot, recipient, current, slotMask);
        if (_popcount32(futureBitmap) > MAX_ACTIVE_SLOTS) revert CapExceeded();
        // Apply
        currentSlate[slot] = recipient;
        currentActiveBitmap = currentActiveBitmap | slotMask;
        emit CharityApplied(slot, recipient);
    } else {
        // Queue branch (current != 0; locked-slot guard already passed → slot in 3..19)
        // Cap check: future bitmap reflects this pending write applied + all other pending edits applied.
        uint32 futureBitmap = _futureBitmapAfter(slot, recipient, current, slotMask);
        if (_popcount32(futureBitmap) > MAX_ACTIVE_SLOTS) revert CapExceeded();
        // Apply (mapping write + bitmap bit set; if bit already set, this is a pending-overwrite — replace value)
        pendingEdit[slot] = recipient;
        pendingEditSet = pendingEditSet | slotMask;
        emit CharityQueued(slot, recipient);
    }
}
```

**Note on the removal-special-case branch:** the implementation above handles the subtle case where `current == 0` AND `recipient == 0` AND a pending edit exists for the slot — it treats this as "cancel the queued add" and clears the pending bit. This IS strictly within the spec — `SlotAlreadyEmpty` only fires when both current AND pending are empty. The cancel-the-queued-add path is a defensible interpretation of "pending overwrite" (D-254-PENDING-01 says "a new `setCharity` call on the same slot overwrites the pending value (no separate cancel API)" — overwriting a pending-add with zero IS the cancel mechanism). Document this in the SUMMARY for Phase 256 + 257 awareness.

**3. Implement `_futureBitmapAfter` internal pure helper.** Computes the would-be future active bitmap for cap-checking. Conservative computation — applies the proposed write to the right of-currentSlate-or-pendingEdit map, then applies all OTHER pending edits as if flush ran:

```solidity
/// @dev Compute the future currentActiveBitmap as if (a) the proposed setCharity write applied,
///      then (b) all pending edits flushed to current slate. Used for cap-check in setCharity.
/// @param slot The slot being written
/// @param recipient The recipient being written (zero = remove)
/// @param current The current value at currentSlate[slot] (read-once cached by caller)
/// @param slotMask uint32(1) << slot (cached by caller)
/// @return future The post-flush currentActiveBitmap if this setCharity call is applied + all pending edits flush
function _futureBitmapAfter(
    uint8  slot,
    address recipient,
    address current,
    uint32 slotMask
) private view returns (uint32 future) {
    // Start from current bitmap.
    future = currentActiveBitmap;

    // Treat the proposed write as a pending entry for the slot, then iterate the union of
    // (pendingEditSet ∪ {slot}) and apply each pending value to the future bitmap. This gives
    // a one-shot model of "apply this proposed setCharity, then flush all pending edits".

    uint32 pSet = pendingEditSet | slotMask;  // include this call's slot in the pending iteration set

    // Iterate the bits of pSet (slots 0..19; bits 20..31 always zero by construction).
    // Worst case: 20 cold SLOADs of pendingEdit[i] (~42000 gas) — bounded by MAX_ACTIVE_SLOTS.
    for (uint8 i; i < MAX_ACTIVE_SLOTS;) {
        uint32 mask_i = uint32(1) << i;
        if ((pSet & mask_i) != 0) {
            address pendingValue;
            if (i == slot) {
                // Use the proposed recipient (overrides any prior pending edit for this slot)
                pendingValue = recipient;
            } else {
                pendingValue = pendingEdit[i];
            }
            if (pendingValue == address(0)) {
                // Pending-remove: clear the bit
                future = future & ~mask_i;
            } else {
                // Pending-add or pending-replace: set the bit
                future = future | mask_i;
            }
        }
        unchecked { ++i; }
    }
}
```

**Note on cap-check semantics (D-254-COUNT-01 alignment):** The future bitmap is the popcount of "current ± all pending edits applied + this proposed write". This matches Plan 03's `activeCountAfterFlush()` view exactly — both compute the same future bitmap, just at different sites. Plan 03 will refactor to share `_futureBitmapAfter` if natural.

**Subtle case — locked-slot first-fill cap interaction:** First-fill of an empty locked slot 0 (currentSlate[0] == 0, slot < LOCKED_SLOTS, locked-slot guard passes because current is zero) — instant-apply branch — cap check via `_futureBitmapAfter` correctly counts this new bit + all other pending edits. No special-casing needed for locked slots in cap math.

**4. Implement `_popcount32` internal pure helper.** Inline-assembly Hamming-weight per Specifics preference:

```solidity
/// @dev Population count (Hamming weight) of a uint32. Constant-gas via inline-asm.
/// @param x The 32-bit value to popcount
/// @return count Number of set bits in `x` (0..32)
function _popcount32(uint32 x) private pure returns (uint8 count) {
    // Hamming-weight algorithm (parallel bit-counting).
    // Reference: https://en.wikipedia.org/wiki/Hamming_weight (popcount_3)
    assembly {
        let v := and(x, 0xFFFFFFFF)
        v := sub(v, and(shr(1, v), 0x55555555))
        v := add(and(v, 0x33333333), and(shr(2, v), 0x33333333))
        v := and(add(v, shr(4, v)), 0x0F0F0F0F)
        v := mul(v, 0x01010101)
        count := and(shr(24, v), 0xFF)
    }
}
```

This is the standard 32-bit popcount; 12 ops, no branches, constant ~30 gas.
