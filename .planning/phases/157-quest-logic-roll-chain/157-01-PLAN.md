**Step C: Add _isLevelQuestEligible** as an internal view function in the LEVEL QUESTS section, after `rollLevelQuest`. Per D-04/D-05/D-06:
```solidity
/// @dev Checks if a player is eligible for the level quest.
///      Requires (levelStreak >= 5 OR any active pass) AND (levelUnits >= 4 this level).
///      Activity gate evaluates first (cheap: same packed word), short-circuiting before
///      the loyalty gate + deity pass fallback.
/// @param player The player address to check.
/// @return True if the player meets both gates.
function _isLevelQuestEligible(address player) internal view returns (bool) {
    uint256 packed = questGame.mintPackedFor(player);

    // Activity gate first (cheap: reads from single packed word)
    uint24 unitsLvl = uint24(packed >> 104);
    if (unitsLvl != questGame.level()) return false;
    uint16 units = uint16(packed >> 228);
    if (units < 4) return false;

    // Loyalty gate: levelStreak >= 5 OR any active pass
    uint24 streak = uint24(packed >> 48);
    if (streak >= 5) return true;

    // Whale/lazy pass from mintPacked_
    uint24 frozen = uint24(packed >> 128);
    uint8 bundle = uint8((packed >> 152) & 0x3);
    if (frozen > 0 && bundle != 0) return true;

    // Deity pass fallback (separate SLOAD)
    return questGame.deityPassCountFor(player) > 0;
}
```
Bit positions: levelStreak at 48, unitsLevel at 104, levelUnits at 228, frozenUntilLevel at 128, whaleBundleType at 152 (per BitPackingLib.sol). Activity gate first per D-06.

```solidity
///      MINT_BURNIE targets 10 tickets, MINT_ETH targets mintPrice * 10,
///      LOOTBOX and DEGENERETTE_ETH target mintPrice * 20,
///      BURNIE-denominated types target 20,000 BURNIE.

function _levelQuestTargetValue(uint8 questType, uint256 mintPrice) internal pure returns (uint256) {
```

**Step E: Add _handleLevelQuestProgress** after `_levelQuestTargetValue`. Per D-08/D-09/D-10/D-11:
```solidity
/// @dev Shared level quest progress handler called by each of the 6 handlers.
///      Reads the level quest type for the current level, checks if it matches the
///      handler's quest type, validates level boundary (resets stale data), checks
///      eligibility, accumulates progress, and triggers the completion flow
///      (creditFlip + event) when the target is met.
/// @param player The player earning progress.
/// @param handlerQuestType The quest type this handler tracks.
/// @param delta The progress delta (units match quest type).
/// @param mintPrice Current mint price in wei (for ETH-based targets).
function _handleLevelQuestProgress(
    address player,
    uint8 handlerQuestType,
    uint256 delta,
    uint256 mintPrice
) internal {
    uint24 currentLevel = questGame.level();
    uint8 lqType = levelQuestType[currentLevel];

    // Guard: at level 0 with type 0, there is no level quest (roll happens on transition TO a level).
    // lqType == 0 with currentLevel > 0 means no quest rolled yet for this level.
    if (lqType == 0 && currentLevel != 0) return;
    if (currentLevel == 0) return;

    // Check handler type match
    if (lqType != handlerQuestType) return;

    uint256 packed = levelQuestPlayerState[player];
    uint24 questLevel = uint24(packed);

    // Level-boundary invalidation: reset stale data from previous levels
    if (questLevel != currentLevel) {
        packed = uint256(currentLevel);
    }

    // Already completed this level's quest
    if ((packed >> 152) & 1 == 1) return;

    // Eligibility check (1-2 SLOADs via questGame)
    if (!_isLevelQuestEligible(player)) return;

    uint128 progress = uint128(packed >> 24);
    progress = _clampedAdd128(progress, delta);

    uint256 target = _levelQuestTargetValue(lqType, mintPrice);
    if (uint256(progress) >= target) {
        // Completion: set completed bit 152, persist progress
        packed = uint256(currentLevel)
               | (uint256(progress) << 24)
               | (uint256(1) << 152);
        levelQuestPlayerState[player] = packed;
        IBurnieCoinflip(ContractAddresses.COINFLIP).creditFlip(player, 800 ether);
        emit LevelQuestCompleted(player, currentLevel, lqType, 800 ether);
    } else {
        // Progress update only
        packed = uint256(currentLevel)
               | (uint256(progress) << 24);
        levelQuestPlayerState[player] = packed;
    }
}
```
Key design points:
- Level 0 guard: `currentLevel == 0` always returns (no quest rolled at level 0 since roll happens on transition TO a level).
- `lqType == 0 && currentLevel > 0`: means MINT_BURNIE (type 0) was never rolled for this level (default mapping value). Returns early.
- Level boundary invalidation per D-11: if questLevel != currentLevel, reset packed to just the level (zero progress, not completed).
- Completion per D-08: sets bit 152, calls `creditFlip(player, 800 ether)` on BurnieCoinflip, emits `LevelQuestCompleted`.
- Reward handled internally (no return value) -- simplifies Phase 158 handler integration.

```solidity
function getPlayerLevelQuestView(address player)
    external
    view
    override
    returns (uint8 questType, uint128 progress, uint256 target, bool completed, bool eligible)
{
    uint24 currentLevel = questGame.level();
    questType = levelQuestType[currentLevel];

    uint256 packed = levelQuestPlayerState[player];
    uint24 questLevel = uint24(packed);

    // If player's stored level mismatches current, treat as fresh (zero progress, not completed)
    if (questLevel == currentLevel) {
        progress = uint128(packed >> 24);
        completed = (packed >> 152) & 1 == 1;
    }
    // else: progress = 0, completed = false (default)

    target = _levelQuestTargetValue(questType, questGame.mintPrice());
    eligible = _isLevelQuestEligible(player);
}
```
