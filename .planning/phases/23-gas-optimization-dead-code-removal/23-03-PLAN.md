const fs = require('fs');
const path = require('path');
const artifactDirs = [
  'artifacts/contracts',
];
function walk(dir) {
  const files = fs.readdirSync(dir, {withFileTypes:true});
  for (const f of files) {
    const p = path.join(dir, f.name);
    if (f.isDirectory()) walk(p);
    else if (f.name.endsWith('.json') && !f.name.endsWith('.dbg.json')) {
      const data = JSON.parse(fs.readFileSync(p));
      if (data.deployedBytecode && data.deployedBytecode.length > 2) {
        const bytes = (data.deployedBytecode.length - 2) / 2;
        if (bytes > 100) console.log(f.name.replace('.json','') + ': ' + bytes + ' bytes (' + (bytes/24576*100).toFixed(1) + '% of limit)');
      }
    }
  }
}
walk('artifacts/contracts');
"
