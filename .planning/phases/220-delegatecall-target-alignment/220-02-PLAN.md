# Naming exceptions: UPPER_SNAKE segment → CamelCase that the default transform can't derive.
# Format: "SNAKE_FRAGMENT:CamelFragment"
declare -A NAMING_EXCEPTIONS=(
  [GAMEOVER]=GameOver
)

constant_to_iface() {
  local c="$1"                                       # GAME_BOON_MODULE
  local stripped="${c#GAME_}"                        # BOON_MODULE
  stripped="${stripped%_MODULE}"                     # BOON
  # Check exception map first
  if [[ -n "${NAMING_EXCEPTIONS[$stripped]:-}" ]]; then

    return 0
  fi
  # Default: UPPER_SNAKE → CamelCase
  local camel
  camel=$(printf '%s' "$stripped" | awk 'BEGIN{FS="_"; OFS=""} { for(i=1;i<=NF;i++) $i=toupper(substr($i,1,1)) tolower(substr($i,2)); print }')

3. **Add the preflight function**:
```bash
# Known dead constants — declared but not expected to have an interface or callers.
# Remove from this list when Phase 223 consolidates the module universe.
DEAD_CONSTANTS=(
  GAME_ENDGAME_MODULE
)

is_dead_constant() {
  local c="$1"
  for dead in "${DEAD_CONSTANTS[@]}"; do
    [[ "$c" == "$dead" ]] && return 0
  done
  return 1
}

validate_mapping() {
  local constants interfaces
  constants=$(grep -oE 'GAME_[A-Z_]+_MODULE' contracts/ContractAddresses.sol | sort -u)
  interfaces=$(grep -oE 'interface IDegenerusGame[A-Za-z]+Module' contracts/interfaces/IDegenerusGameModules.sol \
    | awk '{print $2}' | sort -u)

  local mapping_fails=0

  # Every LIVE constant must have a matching interface
  while IFS= read -r c; do
    [[ -z "$c" ]] && continue
    if is_dead_constant "$c"; then
      printf "%bDEAD%b %-30s known-dead constant (no interface expected)\n" "$YELLOW" "$NC" "$c"
      continue
    fi
    local expected
    expected=$(constant_to_iface "$c")
    if printf '%s\n' "$interfaces" | grep -qx "$expected"; then
      :  # silent OK for individual mapping row; per-site loop will print detail
    else
      printf "%bMAP_FAIL%b %-30s expected interface %s not found in IDegenerusGameModules.sol\n" "$RED" "$NC" "$c" "$expected"
      mapping_fails=$((mapping_fails + 1))
    fi
  done <<< "$constants"

  # Every interface must have a matching constant (reverse direction)
  while IFS= read -r i; do
    [[ -z "$i" ]] && continue
    local expected
    expected=$(iface_to_constant "$i")  # function from Plan 220-01
    if printf '%s\n' "$constants" | grep -qx "$expected"; then
      :
    else
      printf "%bMAP_FAIL%b %-40s expected constant %s not found in ContractAddresses.sol\n" "$RED" "$NC" "$i" "$expected"
      mapping_fails=$((mapping_fails + 1))
    fi
  done <<< "$interfaces"

  if [[ $mapping_fails -gt 0 ]]; then
    printf "%bFAIL%b interface ↔ address map has %d mismatch(es) — universe is inconsistent\n" "$RED" "$NC" "$mapping_fails"
    exit 1
  fi
  printf "%bOK%b interface ↔ address map: %d LIVE pairs validated, %d known-dead constant(s) skipped\n" \
    "$GREEN" "$NC" \
    "$(printf '%s\n' "$constants" | wc -l | tr -d ' ')" \
    "${#DEAD_CONSTANTS[@]}"
}
```
